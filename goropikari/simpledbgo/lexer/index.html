    // TKeyword is keyword token type.
    TKeyword

    // TInt32 is int32 token type.
    TInt32

    // TString is string token type.
    TString

    TIdentifier

    // TStar is star token type.
    TStar

    // TEqual is equal token type.
    TEqual

    TComma
